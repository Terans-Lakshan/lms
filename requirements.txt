# GeoLMS - Project Requirements
# ================================
# This is a Node.js project. Dependencies are managed via npm (package.json).
# Run `npm install` inside /server and /client to install all packages.
#
# For a full setup guide, see: docs/SETUP.md
# ================================


# ──────────────────────────────────────────────
# SYSTEM REQUIREMENTS
# ──────────────────────────────────────────────

Node.js          >= 14.0.0   (https://nodejs.org)
npm              >= 6.0.0    (bundled with Node.js)
MongoDB          >= 4.4      (local) or MongoDB Atlas (cloud)
Git              >= 2.0      (https://git-scm.com)


# ──────────────────────────────────────────────
# OPTIONAL EXTERNAL SERVICES
# ──────────────────────────────────────────────

AWS S3           -- File / material uploads
Google OAuth 2.0 -- Social login (sign in with Google)
Nodemailer / SMTP -- Email verification & password reset

# ──────────────────────────────────────────────
# ENVIRONMENT VARIABLES REQUIRED
# ──────────────────────────────────────────────

# server/.env
PORT                    = 3000
NODE_ENV                = development | production
MONGO_URI               = mongodb://... or mongodb+srv://...
JWT_SECRET              = <random 32+ char string>
AWS_ACCESS_KEY_ID       = <from AWS IAM>
AWS_SECRET_ACCESS_KEY   = <from AWS IAM>
AWS_S3_BUCKET           = <bucket name>
AWS_REGION              = us-east-1
CORS_ORIGIN             = http://localhost:5174
GOOGLE_CLIENT_ID        = <from Google Cloud Console>
GOOGLE_CLIENT_SECRET    = <from Google Cloud Console>
SENDGRID_API_KEY        = <from SendGrid>  (optional)

# client/.env
VITE_API_BASE_URL       = http://localhost:3000
VITE_APP_NAME           = GeoLMS


# ──────────────────────────────────────────────
# QUICK START
# ──────────────────────────────────────────────

# 1. Clone repo
#    git clone <repo-url> && cd GeoLMS

# 2. Install backend
#    cd server && npm install

# 3. Install frontend
#    cd ../client && npm install

# 4. Configure env files (see above)

# 5. Start backend (port 3000)
#    cd server && npm run dev

# 6. Start frontend (port 5174)
#    cd client && npm run dev
